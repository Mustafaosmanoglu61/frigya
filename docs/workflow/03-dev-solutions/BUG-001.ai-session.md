+  function _recalcTotals() {
+    const rows = [...document.querySelectorAll('#transactions-table tbody tr[data-symbol]')]
+      .filter(r => r.style.display !== 'none');
+    const num = (sel) => rows.reduce((a, r) =>
+      a + parseFloat(r.querySelector(sel)?.dataset.raw || 0), 0);
+    const proceeds = num('.col-proceeds');
+    const cost     = num('.col-cost');
+    const pnls     = rows.map(r => parseFloat(r.dataset.pnl || 0));
+    const profit   = pnls.filter(v => v >= 0).reduce((a, b) => a + b, 0);
+    const loss     = pnls.filter(v => v <  0).reduce((a, b) => a + Math.abs(b), 0);
+    const wins     = pnls.filter(v => v >= 0).length;
+    const net      = proceeds - cost;
+    const f = v => '$' + v.toLocaleString('en-US', {minimumFractionDigits: 2, maximumFractionDigits: 2});
+    const set = (k, v) => { const e = document.querySelector(`[data-kpi="${k}"]`); if (e) e.textContent = v; };
+    set('proceeds', f(proceeds));  set('cost', f(cost));
+    set('profit', f(profit));      set('loss', f(loss));
+    set('net-pnl', f(net));
+    set('basari', rows.length ? (wins / rows.length * 100).toFixed(1) + '%' : '—');
+    const tf = document.querySelector('[data-toplam="net-pnl"]'); if (tf) tf.textContent = f(net);
+  }
